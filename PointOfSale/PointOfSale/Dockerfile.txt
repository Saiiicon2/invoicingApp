FROM mcr.microsoft.com/dotnet/sdk:8.0 AS build
WORKDIR /src

# Copy solution and project files
COPY PointOfSale.sln .
COPY PointOfSale/*.csproj ./PointOfSale/
COPY PointOfSale.Business/*.csproj ./PointOfSale.Business/
COPY PointOfSale.Data/*.csproj ./PointOfSale.Data/
COPY PointOfSale.Model/*.csproj ./PointOfSale.Model/

RUN dotnet restore

# Copy everything else
COPY . .

RUN dotnet publish PointOfSale/PointOfSale.csproj -c Release -o /app/publish

FROM mcr.microsoft.com/dotnet/aspnet:8.0 AS final
WORKDIR /app

ENV ASPNETCORE_URLS=http://0.0.0.0:${PORT}

COPY --from=build /app/publish .

ENTRYPOINT ["dotnet", "PointOfSale.dll"]
